Preview Tool: {{ engine }}

Output

{{ output }}

Examples

{% if engine in ["jinja2", "twig"] %}
  • {{ "{{7*7}}" }} → 49
  • {{ "{{'ab' ~ 'cd'}}" }} → abcd
  • {{ "{{'abc'|upper}}" }} → ABC
{% elif engine == "freemarker" %}
  • ${"{7*7}"} → 49
  • ${"\"ab\" + \"cd\""} → abcd
  • ${"\"abc\"?upper_case"} → ABC
{% elif engine == "velocity" %}
  • #set($x=7*7)$x → 49
  • #set($s="ab" + "cd")$s → abcd
  • #set($t="abc")$t.toUpperCase() → ABC
{% elif engine == "mustache" %}
  • {{ "{{#if 1}}OK{{/if}}" }} → OK
  • {{ "{{#unless false}}OK{{/unless}}" }} → OK
  • {{ "{{#with 'abc'}}{{this}}{{/with}}" }} → abc
{% elif engine == "os" %}
  • echo $((7*7)) → echo 49
  • TOASTI_OS_$((123*456)) → TOASTI_OS_56088
{% endif %}
